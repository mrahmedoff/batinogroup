// Firebase Storage Security Rules
// Apply these rules in Firebase Console

// For Development (anyone can upload):
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read, write: if true;
    }
  }
}

// For Production (only authenticated users):
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /logos/{fileName} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
